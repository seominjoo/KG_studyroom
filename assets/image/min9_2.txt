        function newlotto() {
            let numbers = new Array;
            
            for (let i = 0; i < 7; i++) {
                numbers[i] = Math.floor(Math.random() * 45 + 1); 
                for(j = 0; j < i; j++){
                    if(numbers[i] === numbers[j]) {
                        i--;
                        break;
                    }
                }
            }
            for (let index = 1; index <= 7; index++) {
                str = document.getElementById('lotto');
                str.innerHTML += `<p>${numbers[index-1]}</p>`;
            }
        }
